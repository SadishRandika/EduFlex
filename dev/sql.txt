CREATE TABLE Student (
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    gender VARCHAR(10),
    dob DATE,
    email VARCHAR(255),
    admission_date DATE,
    class VARCHAR(50)
);
